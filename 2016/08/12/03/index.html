public static void printmax(double... number){
    if(number.length == 0) System.out.println("error!");


    double maxx = number[0];

    for(int i=1;i<number.length;i++){
        maxx = Math.max(maxx , number[i]);
    }

    System.out.println(maxx);
}

class MyComparator implements Comparator<Integer>{

@Override
public int compare(Integer arg0, Integer arg1) {
    if(arg0 < arg1) return -1; //貌似要想达到什么样子的排序就是返回-1，这样好记忆些
    else if(arg0 == arg1) return 0;//0是不变位置
    else return 1;//返回负值就会对调
}

}
